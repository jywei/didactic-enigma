def self.stack
  temp   = all.dup.map {|log| log.count = 1 ; log }

  result = []
  temp.each do |log|
    duplicate = result.select {|n| 
      n[:name] == log[:name] && n[:path] == log[:path] && n[:message] == log[:message]
    }.first

    duplicate ? duplicate.count += 1 : result << log
  end

  result
end
