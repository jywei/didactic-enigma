def general_info(order, sport_name)
  self.order_id    = order.id
  self.result_flag = if order.stake == 0
                       "draw"
                     elsif order.stake > 0
                       "win"
                     else
                       "lose"
                     end
  self.sport_name  = sport_name
end

def profit_splitting(user_id, sport_name, order)
  sport_name = "us_baseball" if sport_name.include?('baseball')
  ancestorz_arr = User::Ancestor.upperlines(user_id).attributes.values[1..-1]
  oppos = User::Allotter.where("name = ? AND user_id in (?)", sport_name, ancestorz_arr).order(user_id: :desc).pluck(:oppo).map(&:to_f)
  unless oppos.empty?
    ancestor_shares = share_interval(oppos)
  end
  share_cut_update(order.stake, ancestor_shares)
  general_info(order, sport_name)
  save
  order.update!(split: true)
end

def share_cut_update(stake, shares)
  self.member            = stake
  self.admin             = stake * shares[7]
  self.moderator         = stake * shares[6]
  self.supervisor        = stake * shares[5]
  self.director          = stake * shares[4]
  self.major_shareholder = stake * shares[3]
  self.shareholder       = stake * shares[2]
  self.general_agent     = stake * shares[1]
  self.agent             = stake * shares[0]
end

def share_interval(oppos)
  tmp = []
  while(oppos.length != 1)
    tmp << oppos[0].to_f if oppos.length == 8
    tmp << (oppos[1].to_f - oppos[0].to_f).round(3)
    oppos.shift
  end
  tmp
end
