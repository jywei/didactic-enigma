def logging_error(args, action, e, uuid = nil)
  user_id = current_user.try(:id)
  Log::Error.create!(
    name:      e.class.name,
    user_id:   user_id,
    path:      "#{channel}##{action}",
    uuid:      uuid,
    message:   e.message,
    params:    args,
    backtrace: e.backtrace,
    data:      []
  )
end
