def assign_ot_and_type(ot_ids, type)
  types = basics.dup
  ot_ids.each_with_index do |ot, i|
    types[i][:ot]      = ot.to_i
    types[i][:ot_type] = type
  end
  types
end

def basketball_three_way?(ot_id, tx_spid = nil)
  return false if tx_spid.nil?
  tx_spid.to_i == 3 && %w(2 21 237).include?(ot_id.to_s)

def name(ot_id, head, tx_spid = nil)
  data.each do |offer|
    next unless offer[:ot].to_i == ot_id
    if basketball_three_way?(ot_id, tx_spid)
      return 'ml'
    elsif point?(ot_id) && [1.5, -1.5].include?(head.to_f.round(2))
      return 'one_win'
    else
      return offer[:ot_name]
    end
  end
  nil
end

def type(ot_id)
  if full?(ot_id)
    'full'
  elsif ht?(ot_id)
    'ht'
  elsif hf?(ot_id)
    'hf'
  end
end

def type_to_ch(name)
  case name
  when 'full'
    '全場'
  when 'ht'
    '上半場'
  when 'hf'
    '下半場'
  else
    name
  end
end

def type_to_i(type)
  case type
  when 'full' then 0
  when 'ht'   then 1
  when 'hf'   then 2
  when 'q1'   then 3
  when 'q2'   then 4
  when 'q3'   then 5
  when 'q4'   then 6
  end
end
