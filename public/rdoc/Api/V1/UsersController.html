def settings
  @response.message =
    if check_parameter('settings', params) == false
      '使用者ID不可空白'

      bank = User.find_by(id: params[:user_id])

      '成功'
    end

  render json: @response.to_render
end

def sign_up
  # apiRoute: '/api/v1/sign_up?',
                # method: 'POST',

                # params: {
                #           username: id,
                #           password: pw,
                #           bank_id: user.id,
                #           identity: identity,
                #           token: user.token
                #         }
  @response.message =
    if check_parameter('sign_up', params) == false
      '帳號或密碼不可空白'
    elsif User.find_by(username: params[:username])
      '使用者名稱重複'
    else
      bank = User.find_by(id: params[:bank_id].to_i)
      user = {
        username:           params[:username],
        encrypted_password: params[:password],
        email:              params[:username],
        total_quota:        100,
        bank_id:            params[:bank_id].to_i,
        tier:               bank.nil? ? 0 : (bank.try(:tier).to_i + 1),
        access_token:       User.generate_token
      }
      user[:identity] = 'assist' && user[:fork_id] = bank.id && user[:tier] = bank.tier.to_i if params[:identity] == 'assist'
      user = User.create(user)
      update_ancestors(user) if user.tier == 8

      @response.succeed
      @response.set_column(:user, id: user.id, name: user.username, bank_id: user.bank_id, token: user.access_token)
      '註冊成功'
    end

  render json: @response.to_render
end

def update_allotters
  @response.message =
    if check_parameter('update_allotters', params) == false
      '傳入資訊不可空白'
    elsif User.find_by(id: params[:target_id]).nil?
      '找不到修改使用者'
    else
      origin_allotters = User::Allotter.find_by(user_id: params[:target_id], name: params[:category_name])
      origin_allotters.try(:update, oppo: params[:value].to_f) if origin_allotters[:oppo].to_f >= params[:value].to_f
      @response.succeed
      '更新成功'
    end

  render json: @response.to_render
end
