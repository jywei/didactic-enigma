def downlines(heirs, order_ancestors)
  tier_names = tier_name(heirs.tier, :en)
  tier_names.each_with_index.map do |name, index|
    if name == 'member'
    {
      name: '會員',
      vigorish: 0,
      result: order_ancestors.map(&:member).sum.to_f
    }
    else
    {
      name: downline_names[(heirs.tier+index)][Lang[:ch]],
      vigorish: 0,
      unsplit_stake: 0,
      result: order_ancestors.map(&:"#{tier_names[index]}").sum.to_f
    }
    end
  end #each_with_index
end

def element(heir, orders, order_ancestors)
  {
    id:                 heir.id,
    name:               heir.username,
    allotters:          allotters_report(heir.id),
    orders_count:       orders.count,
    amount:             orders.map(&:amount).sum,
    valid_amount:       order_ancestors.map(&:member).sum.to_f,
    unposted_amount:    0,
    downline:           UserReport.downlines(heir, order_ancestors)
  }

def filter_params(params)
  @sport_ids = params[:sport_id].present? ? params[:sport_id].split(",") : nil  
  @type_ids  = params[:type_id].present? ? params[:type_id].split(",") : nil
  @time      = params[:start_date].present? ? Time.parse(params[:start_date])..Time.parse(params[:end_date]) : nil
end

def params_valid?
  @messages = []
  @messages << "找不到對應ID #{params[:id]} 的user" if @user.nil?
  @messages << "未帶入sport_id參數" if @sport_ids.blank?
  @messages << "未帶入type_id參數" if @type_ids.blank?
  @messages << "未帶入start_time或end_time參數" if @time.blank?
  @message.blank?
end
